推薦組合方案
💡 在 LiteLLM 設定多個 model entry，Claude Code 就能用不同的 model name 呼叫不同模型，不需重新載入。
litellm_config.yaml 
# LiteLLM Proxy 設定 — 48GB GPU 完整版
# 支援 NVIDIA Nemotron + Google Gemma 模型自由切換

model_list:

  # ─── NVIDIA Nemotron 系列 ────────────────────────────────

  # 日常快速 coding（3.6B active，約 6GB VRAM）
  - model_name: nemotron-nano
    litellm_params:
      model: ollama/nemotron-3-nano
      api_base: http://localhost:11434

  # 主力開發 / 複雜任務（12B active，約 16GB VRAM）⭐ 推薦
  - model_name: nemotron-super
    litellm_params:
      model: ollama/nemotron-3-super
      api_base: http://localhost:11434

  # 旗艦通用（70B，約 40GB VRAM）
  - model_name: nemotron-70b
    litellm_params:
      model: ollama/nemotron
      api_base: http://localhost:11434

  # ─── Google Gemma / CodeGemma 系列 ──────────────────────

  # 輕量快速補全（約 6GB VRAM）
  - model_name: codegemma-7b
    litellm_params:
      model: ollama/codegemma:7b
      api_base: http://localhost:11434

  # 平衡 coding + 解釋（約 8GB VRAM）⭐ 推薦
  - model_name: gemma2-9b
    litellm_params:
      model: ollama/gemma2:9b
      api_base: http://localhost:11434

  # 高品質通用（約 18GB VRAM）⭐ 推薦
  - model_name: gemma2-27b
    litellm_params:
      model: ollama/gemma2:27b
      api_base: http://localhost:11434

  # ─── Claude Code 預設入口（對應最強 coding 模型）───────
  - model_name: claude-sonnet-4-5
    litellm_params:
      model: ollama/nemotron-3-super
      api_base: http://localhost:11434

general_settings:
  master_key: "sk-local-dev-2025"
  drop_params: true      # 忽略 Anthropic 專屬參數，避免報錯

litellm_settings:
  set_verbose: false
  request_timeout: 600     # 大模型 timeout 設長一點
~/.zshrc 或 ~/.bashrc — 環境變數 
# ─── Claude Code 指向本地 LiteLLM Proxy ───────────────
export ANTHROPIC_BASE_URL="http://localhost:4000"
export ANTHROPIC_API_KEY="sk-local-dev-2025"

# 套用設定
# source ~/.zshrc
docker-compose.yml — 一鍵啟動（推薦） 
services:

  ollama:
    image: ollama/ollama
    ports:
      - "11434:11434"
    volumes:
      - ollama_data:/root/.ollama
    deploy:
      resources:
        reservations:
          devices:
            - capabilities: [gpu]
    restart: unless-stopped

  litellm:
    image: ghcr.io/berriai/litellm:main-latest
    ports:
      - "4000:4000"
    volumes:
      - ./litellm_config.yaml:/app/config.yaml
    command: ["--config", "/app/config.yaml", "--port", "4000"]
    depends_on:
      - ollama
    restart: unless-stopped

volumes:
  ollama_data:
